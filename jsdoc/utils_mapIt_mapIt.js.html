/**
 * @module MapIt
 * @description Заполняет объект на основе аргументов.
 * @param {Array} keys - массив ключей
 * @param {Array} values - массив значений
 * @returns {Object} result
 */
module.exports = getMap = (keys, values) => {
	let result = {};

	keys.forEach((key, i) => {
		result[key] = values[i];
	});

	return result;
};
